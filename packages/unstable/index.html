quadletman package repository

Native RPM and DEB packages for quadletman.

Fedora / RHEL / AlmaLinux / Rocky Linux

# Import the signing key
sudo rpm --import https://mikkovihonen.github.io/quadletman/packages/unstable/gpg-key.asc

# Add the repository
sudo tee /etc/yum.repos.d/quadletman.repo <<'EOF'
[quadletman]
name=quadletman
baseurl=https://mikkovihonen.github.io/quadletman/packages/unstable/rpm/
enabled=1
gpgcheck=1
repo_gpgcheck=1
gpgkey=https://mikkovihonen.github.io/quadletman/packages/unstable/gpg-key.asc
EOF

# Install
sudo dnf install quadletman

Ubuntu / Debian

# Import the signing key
curl -fsSL https://mikkovihonen.github.io/quadletman/packages/unstable/gpg-key.asc \
  | sudo gpg --dearmor -o /etc/apt/keyrings/quadletman.gpg

# Add the repository
echo "deb [signed-by=/etc/apt/keyrings/quadletman.gpg] https://mikkovihonen.github.io/quadletman/packages/unstable/deb/ stable main" \
  | sudo tee /etc/apt/sources.list.d/quadletman.list

# Install
sudo apt update
sudo apt install quadletman

Verify the signing key

# Download and inspect
curl -fsSL $REPO_URL/gpg-key.asc | gpg --show-keys
